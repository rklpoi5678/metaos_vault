## 리액트에서 fetch사용하기
```js
export default async function getFoods() {
	const response = await fetch('....');
	const body = await response.json();
	return body;
}
```

## useEffect로 초기 데이터 가져오기
```jsx
// 무한 루프가 발생시 네트워크에서 일시정지버튼을 누른다.
const handleLoad = async() => {
	const { reviews } = await getReviews();
	setItems(reviews);
};

handleLoad();
```
