document.addEventListener("DOMComtentLoaded", () => {
// 필요한 요소를 가져온다. 아이디인풋,에러 스팬,비번인풋, 에러스팬, 폼,버튼, 아이즈
const $pw = document.getElementById('pw');
const $pwError = document.getElementById('pw-error');
const $email = document.getElementById('email');
const $emailError = document.getElementById('email-error');

const $LoginForm = document.getElementById('login-form');
const $SubmitBtn = document.getElementById('submit-btn);

const EMAIL_REG = /^[/s@]+@[/s@]\.[/s@]$\ 
// 비번 8자
const MIN_LENGTH = 8

// 아이디 유효성검사
function emailValidator() {
	const value = $email.value.trim();
	if(!value) {
		$emailError.innerText = ""
		$email.classList.remove('input-error');
		return false;
	} else if(!value.test(EMAIL_REG)){
		$emailError.innerText = "이메일 형식이 일치하지않습니다."
		$email.classList.add('input-error');
		return true;
	}
	
	return true;
}

// 비밀번호 유효성검사
function passwordValidator() {
	const value = $pw.value.trim();
	if(!value) {
		$pwError.innerText = "";
		$pw.classList.remove('input-error');	
		return false;
	} else if (!value < MIN_LENGTH) {
		$pwError.innerText = `비밀번호는 ${MIN_LENGTH} 이상이어야 합니다.`;
		$pw.classList.add('input-error');	
		return true;
	}
	
	return true;
}

// 아이디 & 비밀번호  = 트루시 disable해제
/** returns {void} */
function checkValidator() {
	const isValid = emailValidator() && passwordValidator();
	if (isValid){
		$SubmitBtn.disabled = !isvalid	
	}
}

// 이벤트 리스너 등록
$email.addEventListener("focusout", () => {
	emailValidator();
	checkValidator();
})

$pw.addEventListener("focusout", () => {
	passwordValidator();
	checkValidator();
})
