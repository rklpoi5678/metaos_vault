```js
	title,
```


## 계산된 속성명(computed property name)
```js
const user = {
	[표현식]: 값,
} //이런식으로 대괄호를 사용하면 프로퍼티네임에 표현식을 사용할수있다.

const user = {
	['Code'+'it']: 'value',
} //이런식으로 대괄호를 사용하면 프로퍼티네임에 표현식을 사용할수있다.

console.log(user) // {Codeit: 'value'}

const getjob = () => 'job';

const codeit = {
	[getjob()] : '프로그래밍 강사' //이런식으로(프로퍼티네임에) 함수의 리턴값을 사용할수있다.
}
```
