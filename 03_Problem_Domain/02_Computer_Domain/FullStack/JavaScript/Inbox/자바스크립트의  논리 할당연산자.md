## 핵심
ECMAScript2021(ES2021)의 나온 기본 로직을 변경하지 않고  일반적인 조건부 할당을 간소화하는 기능이다.

논리 할당자는 
```js
// 옵셔널 체이닝은 논리 할당의 왼쪽에 허용되지 않는다.
//  속성 자체에 대한 참조가 아닌 속성 엑세스의 결과를 반환한다.
user?.settings ||= {};

/** ---- 논리 OR 할당 (||=) ------- */
//fasly인 경우 할당 0,'' 또는 false와 같은 값을 덮어쓰게 됨, 기본값을 설정하는데유용
user.theme ||= 'light';

if(!user.theme) {
	user.theme = 'light';
}

/** ---- 논리 OR 할당 (||=) ------- */
// truthy한 경우 조건부로 값을 업데이트하는 데 유용하다.
user.isLoggedIn && checkPermission(user);

if(user.isLoggedIn) {
	user.isLoggedIn = checkPermission(user);
}

/** ---- 할당한 예제 ------- */
let isEnabled  = true;
isEnabled &&= false;
console.log(isEnabeld);

/** ---- 할당한 예제 ------- */
// 0, false, ''와 같은 유효한 값을 유지한다.
// 또한 의도치 않은 null/undefined할당을 방지한다.
settings.timeout ??= 3000;

if (settings.timeout === null || settings.timeout === undefined) {
	settings.timeout = 3000;
}

/** ---- 사이드 이펙트가 있는 예제 ------- */
let calls = 0;
let obj = { val: 0 };

obj.val ||= ++calls;
console.log(obj.val); // 1, 0은 falsy이기 때문입니다.

obj.val ||= ++calls;
console.log(obj.val); // 여전히 1 두 번째 증분은 절때 실행되지 않는다.

```
