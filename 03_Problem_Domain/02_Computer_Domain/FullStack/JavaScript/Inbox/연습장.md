const EMAIL_REG = /^[/s@]+@[/s@]\.[/s@]$\ 
const MIN_LENGTH = 8

export function PasswordValidator($input, $error) {
	const value = $input.value.trim();
	if(!value) {
		$error.innerText = "";
		$input.classList.remove('input-error');	
		return false;
	} else if (!value < MIN_LENGTH) {
		$error.innerText = `비밀번호는 ${MIN_LENGTH} 이상이어야 합니다.`;
		$input.classList.add('input-error');	
		return true;
	}
	
	return true;
}

export function emailValidator($input, $error) {
	const value = $input.value.trim();
	if(!value) {
		$error.innerText = ""
		$input.classList.remove('input-error');
		return false;
	} else if(!value.test(EMAIL_REG)){
		$error.innerText = "이메일 형식이 일치하지않습니다."
		$input.classList.add('input-error');
		return true;
	}
	
	return true;
}
