## 배열의 splice, sort는 화살표 함수로 하면 안된다.
filter같은 배열을 만드는것들은 화살표 함수를 쓸수있다.

25/09/11
```js
// n 의 짝수의 합
const xs = (n) => {
  Array(n)
    .fill(0)
		// _은 안쓰고 무시해도되는 변수에 _ 라고 쓰는 관용적인 규칙이다.
    .map((_, i) => i + 1)
    .filter((value) => value % 2 === 0)
    .reduce((acc, cur) => acc + cur , 0)
}

console.log(xs(10))
```
